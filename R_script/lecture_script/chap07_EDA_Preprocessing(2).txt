# 3) 결측데이터 처리(0 or 상수 대체)

dataset$price2 = ifelse( !is.na(x), x, 0) # 결측치 0 대체

# 4) 결측데이터 처리(평균 or 중위수 대체)
x <- dataset$price # price vector 생성 
dataset$price3 = ifelse(!is.na(x), x, round(mean(x, na.rm=T), 2) ) # 평균대체
